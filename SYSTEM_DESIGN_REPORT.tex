\subsubsection{State Management Strategy}

\textbf{Pattern:} React Context API + Custom Hooks (No Redux needed at this scale)

\begin{tcolorbox}[colback=accentgreen!10, colframe=accentgreen, boxrule=1.5pt]
\textbf{State Layers:}
\begin{itemize}
    \item \textbf{Global State:} User auth, theme (Context API)
    \item \textbf{Page State:} Filters, search, pagination (useState)
    \item \textbf{Component State:} Form inputs, UI toggles (useState)
    \item \textbf{Server State:} API responses cached locally (useEffect)

\end{tcolorbox}

\textbf{Security Implementation:}
\begin{tcolorbox}[colback=accentred!10, colframe=accentred, boxrule=1.5pt]
\begin{itemize}
    \item \textbf{Password Hashing:} bcryptjs with 10 salt rounds (industry standard)
    \item \textbf{Token Expiry:} 7 days (configurable)
    \item \textbf{Protected Routes:} Middleware validates JWT on every request
    \item \textbf{Email Normalization:} All emails converted to lowercase
    \item \textbf{Rate Limiting:} 100 requests/minute per endpoint
    \item \textbf{CORS:} Strict origin validation
    \item \textbf{Input Validation:} Zod schemas for all endpoints
    \item \textbf{SQL/NoSQL Injection:} Prevented via Mongoose (parameterized queries)
\end{itemize}
\end{tcolorbox}

\subsubsection{Users Collection}

\begin{lstlisting}
db.users.createIndex({ email: 1 }, { unique: true })
db.users.createIndex({ role: 1 })

{
  _id: ObjectId,

  password: String (bcrypt hashed, required),
  role: Enum(ADMIN, WARDEN, GUEST),
  organization: String (auto-populated),
  building: String (optional, indexed),
  metadata: {
    lastLogin: Date,
    loginCount: Number,

  createdAt: Date (indexed),
  updatedAt: Date
}

db.rooms.createIndex({ price: 1 })

db.rooms.createIndex({ text: 'text' }) // Full-text search

{

  type: Enum (Single, Double, Suite, Studio),
  price: Number (min: 0, indexed),
  amenities: Array<String>,

\subsection{Indexing Strategy \& Query Optimization}

\begin{tcolorbox}[colback=accentyellow!10, colframe=accentyellow, boxrule=1.5pt]
\textbf{\textcolor{darkblue}{Query Optimization Techniques:}}
\begin{itemize}
    \item \textbf{Composite Indexes:} \texttt{(university, isAvailable, price)} for common queries
    \item \textbf{Index Intersection:} MongoDB uses multiple indexes for OR queries
    \item \textbf{Covered Queries:} Select only needed fields to avoid document lookups
    \item \textbf{Query Explain:} Regular performance monitoring with explain()
    \item \textbf{Aggregation Pipeline:} For complex queries instead of application-level processing
\end{itemize}
\end{tcolorbox}
